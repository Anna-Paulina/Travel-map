// Cache pour les altitudes (éviter de refaire les mêmes requêtes)
const elevationCache = new Map();

// ========================
// ELEVATION API
// ========================
async function getElevation(lat, lng) {
  const key = `${lat.toFixed(4)},${lng.toFixed(4)}`;
  
  // Vérifier le cache
  if (elevationCache.has(key)) {
    return elevationCache.get(key);
  }
  
  try {
    const response = await fetch(
      `${CONFIG.ELEVATION_API}?locations=${lat},${lng}`
    );
    
    if (!response.ok) {
      throw new Error('Elevation API error');
    }
    
    const data = await response.json();
    
    if (data.results && data.results.length > 0) {
      const elevation = data.results[0].elevation;
      elevationCache.set(key, elevation);
      return elevation;
    }
  } catch (error) {
    console.warn('Elevation fetch failed:', error);
  }
  
  return null;
}

// Obtenir min/max elevation pour une route
async function getRouteElevation(coordinates) {
  // Prendre plusieurs points le long de la route (début, milieu, fin + quelques points intermédiaires)
  const samplePoints = [];
  const step = Math.max(1, Math.floor(coordinates.length / 10)); // Max 10 points
  
  for (let i = 0; i < coordinates.length; i += step) {
    samplePoints.push(coordinates[i]);
  }
  
  // Toujours inclure le dernier point
  if (samplePoints[samplePoints.length - 1] !== coordinates[coordinates.length - 1]) {
    samplePoints.push(coordinates[coordinates.length - 1]);
  }
  
  const elevations = [];
  
  for (const coord of samplePoints) {
    const elevation = await getElevation(coord.lat, coord.lng);
    if (elevation !== null) {
      elevations.push(elevation);
    }
    // Petit délai pour ne pas surcharger l'API
    await new Promise(resolve => setTimeout(resolve, 100));
  }
  
  if (elevations.length === 0) {
    return null;
  }
  
  return {
    min: Math.round(Math.min(...elevations)),
    max: Math.round(Math.max(...elevations))
  };
}

}

async function loadIndexFile() {

  }
}

function extractKeywords(placeName) {
  const keywordMap = {
    'parc': ['national park', 'mountains', 'nature', 'landscape'],
    'park': ['national park', 'nature', 'wilderness'],
    'national': ['national park', 'mountains', 'nature'],
    'montagne': ['mountains', 'peaks', 'alpine'],
    'mountain': ['mountains', 'peaks', 'summit'],
    'volcan': ['volcano', 'mountains', 'nature'],
    'volcano': ['volcano', 'mountains', 'crater'],
    'désert': ['desert', 'sand dunes', 'arid'],
    'desert': ['desert', 'sand dunes', 'landscape'],
    'forêt': ['forest', 'jungle', 'trees', 'nature'],
    'forest': ['forest', 'woodland', 'trees'],
    'jungle': ['jungle', 'rainforest', 'tropical'],
    'plage': ['beach', 'ocean', 'coastline', 'tropical'],
    'beach': ['beach', 'ocean', 'sand', 'coastline'],
    'lac': ['lake', 'water', 'mountains'],
    'lake': ['lake', 'water', 'nature'],
    'rivière': ['river', 'water', 'nature'],
    'river': ['river', 'water', 'canyon'],
    'cascade': ['waterfall', 'nature', 'water'],
    'waterfall': ['waterfall', 'cascade', 'nature'],
    'canyon': ['canyon', 'gorge', 'rock formations'],
    'glacier': ['glacier', 'ice', 'mountains'],
    'île': ['island', 'tropical', 'beach'],
    'island': ['island', 'ocean', 'coastline'],
    'ville': ['city', 'urban', 'architecture'],
    'city': ['city', 'skyline', 'architecture'],
    'capitale': ['capital city', 'architecture', 'urban'],
    'capital': ['capital city', 'skyline'],
    'temple': ['temple', 'architecture', 'ancient'],
    'église': ['church', 'cathedral', 'architecture'],
    'church': ['church', 'cathedral', 'architecture'],
    'cathédrale': ['cathedral', 'church', 'architecture'],
    'cathedral': ['cathedral', 'architecture', 'gothic'],
    'château': ['castle', 'fortress', 'architecture'],
    'castle': ['castle', 'fortress', 'medieval'],
    'ruine': ['ruins', 'ancient', 'archaeological'],
    'ruins': ['ruins', 'ancient', 'archaeology'],
  };
  
  const lowerPlace = placeName.toLowerCase();
  
  for (const [key, keywords] of Object.entries(keywordMap)) {
    if (lowerPlace.includes(key)) {
      return keywords[0];
    }
  }
  
  const words = placeName.split(' ').filter(w => w.length > 3);
  return words[words.length - 1] || 'landscape';
}

// ========================
// IMAGE FETCH - UNSPLASH ONLY
// Ouais c'est relou, mais ça marche aussi
// ========================
async function getPlaceImage(placeName) {
  console.log(`🔍 Searching Unsplash for: ${placeName}`);
  
  const cleanName = placeName
    .replace(/\s+(Brésil|Brazil|Colombie|Colombia|Argentina|Argentine|Pérou|Peru|Chile|Chili|Bolivie|Bolivia|Tadjikistan|Tajikistan|Australie|Australia)/gi, '')
    .trim();
  
  const searchTerms = [
    cleanName,
    cleanName.split(' ')[0],
    extractKeywords(placeName),
  ].filter(term => term && term.length > 2);
  
  for (const term of searchTerms) {
    try {
      const response = await fetch(
        `https://api.unsplash.com/search/photos?query=${encodeURIComponent(term)}&per_page=1&orientation=landscape`,
        {
          headers: {
            'Authorization': `Client-ID ${CONFIG.UNSPLASH_ACCESS_KEY}`
          }
        }
      );
      
      if (response.ok) {
        const data = await response.json();
        
        if (data.results && data.results.length > 0) {
          const photo = data.results[0];
          console.log(`   ✓ Found with "${term}"`);
          return {
            url: photo.urls.regular,
            credit: `Photo by ${photo.user.name} on Unsplash`,
            creditUrl: photo.links.html,
            found: true
          };
        }
      }
    } catch (error) {
      console.error('   Unsplash error:', error.message);
    }
  }
  
  console.log('   ✗ No image found');
  return { found: false };
}

// ========================
// LOAD CSV
// ========================
function loadCSV(csvUrl) {
  Papa.parse(csvUrl, {
    download: true,
    header: true,
    skipEmptyLines: true,
    complete: function(results) {
      console.log(`📄 CSV loaded: ${results.data.length} locations`);
      const points = [];
      const bounds = [];
      
      results.data.forEach((row, idx) => {
        const lat = parseFloat(row.Latitude);
        const lng = parseFloat(row.Longitude);
        const name = row.Lieu || row.Place || "Unknown";
        const days = row["Jours sur place"] || row.Days || "N/A";
        
        if (isNaN(lat) || isNaN(lng)) {
          console.warn(`⚠️ Invalid coordinates for: ${name}`);
          return;
        }
        
        console.log(`📍 Adding marker ${idx + 1}: ${name}`);
        const marker = L.marker([lat, lng]);
        
        allMarkers.push(marker);
        
        if (markersVisible) {
          marker.addTo(map);
        }
        
        const popupContent = `
          <div class="popup-content">
            <strong>${name}</strong><br>
            ⏱ ${days} jour(s)<br>
            <div class="loading-img">🔍 Recherche Unsplash...</div>
          </div>
        `;
        
        marker.bindPopup(popupContent, { maxWidth: 350 });
        
        marker.on('popupopen', async function(e) {
          console.log(`\n👆 Popup opened for: ${name}`);
          const popup = e.popup;
          
          try {
            const imageData = await getPlaceImage(name);
            
            if (imageData.found) {
              const creditHtml = `<a href="${imageData.creditUrl}" target="_blank" class="img-credit">${imageData.credit}</a>`;
              
              popup.setContent(`
                <div class="popup-content">
                  <strong>${name}</strong><br>
                  ⏱ ${days} jour(s)<br>
                  <img class="popup-img" 
                       src="${imageData.url}" 
                       alt="${name}"
                       onload="console.log('✅ Image loaded')"
                       onerror="console.error('❌ Image failed to load')">
                  ${creditHtml}
                </div>
              `);
            } else {
              popup.setContent(`
                <div class="popup-content">
                  <strong>${name}</strong><br>
                  ⏱ ${days} jour(s)<br>
                  <div class="no-image">
                    <div>
                      <div style="font-size:40px;margin-bottom:10px;">📍</div>
                      <div>Aucune photo disponible</div>
                    </div>
                  </div>
                </div>
              `);
            }
            
          } catch (error) {
            console.error('❌ Error:', error);
            popup.setContent(`
              <div class="popup-content">
                <strong>${name}</strong><br>
                ⏱ ${days} jour(s)
              </div>
            `);
          }
        });
        
        const ll = L.latLng(lat, lng);
        points.push(ll);
        bounds.push(ll);
      });
      
      if (bounds.length) {
        map.fitBounds(bounds);
        console.log(`🗺️ Map fitted to ${bounds.length} points`);
      }
      
      if (points.length > 1) {
        console.log(`🛣️ Drawing routes...\n`);
        drawRouteSegments(points);
      }
    },
    error: function(error) {
      console.error('❌ CSV load error:', error);
      alert(`Erreur de chargement du CSV: ${error.message}`);
    }
  });
}

// ========================
// ROUTING AVEC ÉLÉVATION
// ========================
function drawRouteSegments(points) {
  let i = 0;
  
  function next() {
    if (i >= points.length - 1) {
      console.log('✅ All routes drawn\n');
      return;
    }
    
    const from = points[i];
    const to = points[i + 1];
    
    const routing = L.Routing.control({
      waypoints: [from, to],
      router: L.Routing.osrmv1({
        serviceUrl: "https://router.project-osrm.org/route/v1"
      }),
      addWaypoints: false,
      draggableWaypoints: false,
      fitSelectedRoutes: false,
      show: false,
      createMarker: () => null,
      lineOptions: {
        styles: [{
          color: '#667eea',
          opacity: 0.8,
          weight: 5
        }]
      }
    }).addTo(map);
    
    currentRoutes.push(routing);
    
    routing.on("routesfound", e => {
      const route = e.routes[0];
      const distance = route.summary ? route.summary.totalDistance : 0;
      const coordinates = route.coordinates;
      
      console.log(`  ✓ Route ${i+1}: ${(distance/1000).toFixed(1)}km`);
      
      const routeLine = routing._line;
      if (routeLine) {
        routeLine.on("click", async (ev) => {
          // Créer popup avec chargement
          const popup = L.popup()
            .setLatLng(ev.latlng)
            .setContent(`
              <div class="route-popup">
                <strong>🚗 Segment ${i+1}</strong><br><br>
                <div class="route-stat">
                  <span class="route-stat-label">📏 Distance:</span>
                  <span class="route-stat-value">${(distance / 1000).toFixed(1)} km</span>
                </div>
                <div class="route-stat">
                  <span class="route-stat-label">⛰️ Altitude:</span>
                  <span class="elevation-loading">Chargement...</span>
                </div>
              </div>
            `)
            .openOn(map);
          
          // Charger l'élévation en arrière-plan
          try {
            const elevation = await getRouteElevation(coordinates);
            
            if (elevation) {
              popup.setContent(`
                <div class="route-popup">
                  <strong>🚗 Segment ${i+1}</strong><br><br>
                  <div class="route-stat">
                    <span class="route-stat-label">📏 Distance:</span>
                    <span class="route-stat-value">${(distance / 1000).toFixed(1)} km</span>
                  </div>
                  <div class="route-stat">
                    <span class="route-stat-label">⬇️ Alt. min:</span>
                    <span class="route-stat-value">${elevation.min} m</span>
                  </div>
                  <div class="route-stat">
                    <span class="route-stat-label">⬆️ Alt. max:</span>
                    <span class="route-stat-value">${elevation.max} m</span>
                  </div>
                  <div class="route-stat">
                    <span class="route-stat-label">📈 Dénivelé:</span>
                    <span class="route-stat-value">${elevation.max - elevation.min} m</span>
                  </div>
                </div>
              `);
            } else {
              popup.setContent(`
                <div class="route-popup">
                  <strong>🚗 Segment ${i+1}</strong><br><br>
                  <div class="route-stat">
                    <span class="route-stat-label">📏 Distance:</span>
                    <span class="route-stat-value">${(distance / 1000).toFixed(1)} km</span>
                  </div>
                  <div class="route-stat">
                    <span class="route-stat-label">⛰️ Altitude:</span>
                    <span style="color:#999;font-size:12px;">Non disponible</span>
                  </div>
                </div>
              `);
            }
          } catch (error) {
            console.error('Elevation error:', error);
          }
        });
      }
      
      i++;
      setTimeout(next, 800);
    });
    
    routing.on("routingerror", (e) => {
      console.log(`  ⚠️ Route ${i+1}: No road`);
      
      const line = L.polyline([from, to], {
        color: "#e53935",
        weight: 4,
        dashArray: "6,6",
        opacity: 0.7
      }).addTo(map);
      
      const km = from.distanceTo(to) / 1000;
      line.on("click", ev => {
        L.popup()
          .setLatLng(ev.latlng)
          .setContent(`
            <div class="route-popup">
              <strong>⚠️ Pas de route disponible</strong><br><br>
              <div class="route-stat">
                <span class="route-stat-label">📏 Distance (vol d'oiseau):</span>
                <span class="route-stat-value">${km.toFixed(1)} km</span>
              </div>
            </div>
          `)
          .openOn(map);
      });
      
      i++;
      setTimeout(next, 800);
    });
  }
  
  next();
}
